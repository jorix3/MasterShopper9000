001import com.dropbox.core.DbxException;
002import com.dropbox.core.DbxRequestConfig;
003import com.dropbox.core.v2.DbxClientV2;
004
005import java.io.*;
006
007/**
008 * DropBoxManager controls DropBox connection
009 * 
010 * @author      Jyri Virtaranta jyri.virtaranta@cs.tamk.fi
011 * @version     2017.12.02
012 * @since       1.8
013 */
014public class DropBoxManager {
015    private final String ACCESS_TOKEN = "ATeMJijlPUQAAAAAAAAIW72Kz5qt8Kdb"
016                                    + "3ldV6YDLwBvSSf6c6Umf6MuLhsbezGNB";
017    private DbxClientV2 client;
018
019    /**
020     * Instantiates a new DropBox manager with default properties.
021     */
022    public DropBoxManager() {
023        DbxRequestConfig config = DbxRequestConfig
024                                    .newBuilder("MasterShopper9000/1.0")
025                                    .build();
026        client = new DbxClientV2(config, ACCESS_TOKEN);
027    }
028
029    /**
030     * Instantiates a new DropBox manager with custom properties.
031     *
032     * @param  accessToken  identifies DropBox account to connect to
033     */
034    public DropBoxManager(String accessToken) {
035        DbxRequestConfig config = DbxRequestConfig
036                                    .newBuilder("MasterShopper9000/1.0")
037                                    .build();
038        client = new DbxClientV2(config, accessToken);
039    }
040
041    /**
042     * Saves file to DropBox.
043     *
044     * @param  file  the file to save
045     */
046    public void save(File file) {
047        try (InputStream in = new FileInputStream(file)) {
048            String fileName = file.getName();
049            client.files().uploadBuilder("/" + fileName).uploadAndFinish(in);
050        } catch (DbxException | IOException e) {
051            e.printStackTrace();
052        }
053    }
