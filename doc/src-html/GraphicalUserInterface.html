219    /**
220     * Modify shoppingList based on text fields.
221     *
222     * <p>If text field 'name' is not on the list adds new items.
223     * If it is on the list and given boolean is true it increases existing items amount.
224     * If it is on the list and given boolean is false it decreases existing items amount.</p>
225     *
226     * @param  isSubtract  boolean determines if adding or subtracting
227     */
228    private void modifyShoppingList(boolean isSubtract) {
229        if (!amountField.getText().isEmpty()
230                && !nameField.getText().isEmpty()) {
231            int amount;
232
233            try {
234                amount = Integer.parseInt(amountField.getText().trim());
235            } catch (NumberFormatException e) {
236                e.printStackTrace();
237                return;
238            }
239
240            if (isSubtract) {
241                amount = 0 - amount;
242            }
243
244            String name = nameField.getText().trim();
245            shoppingListApp.addToList(new ShoppingItem(amount, name));
246        }
247
248        updateTable();
249    }
250
251    /**
252     * Update tableView contents to reflect shoppingList.
253     */
254    private void updateTable() {
255        shoppingList = shoppingListApp.getShoppingList();
256        table.getItems().clear();
257
258        for (int i = 0; i < shoppingList.size(); i++) {
259            table.getItems().add(shoppingList.get(i));
260        }
261    }
262
263    /**
264     * Create tableView for shoppingItems.
265     *
266     * @return  the created tableView
267     */
268    private TableView<ShoppingItem> createTable() {
269        TableView<ShoppingItem> table = new TableView<>();
270        TableColumn<ShoppingItem, Number> amountCol;
271        TableColumn<ShoppingItem, String> nameCol;
272
273        amountCol = new TableColumn<>("Amount");
274        nameCol = new TableColumn<>("Name");
275
276        amountCol.setCellValueFactory(new PropertyValueFactory<>("amount"));
277        nameCol.setCellValueFactory(new PropertyValueFactory<>("name"));
278
279        amountCol.prefWidthProperty().bind(table.widthProperty().multiply(0.3));
280        nameCol.prefWidthProperty().bind(table.widthProperty().multiply(0.7));
281
282        amountCol.setResizable(false);
283        nameCol.setResizable(false);
284
285        table.getColumns().add(amountCol);
286        table.getColumns().add(nameCol);
287
288        return table;
289    }
