001import javafx.application.Application;
002
003/**
004 * App contains main method for MasterShopper9000
005 * 
006 * @author      Jyri Virtaranta jyri.virtaranta@cs.tamk.fi
007 * @version     2017.11.05
008 * @since       1.8
009 */
010public class App {
011    /**
012     * Launches either CLI or GUI based on command line parameters.
013     *
014     * <p>If not given parameters tries to launch GUI,
015     * if that fails tries to launch CLI.
016     * If given parameter "gui" will not try to launch CLI if GUI fails.
017     * If given parameter "cli" will not try to launch GUI if CLI fails.</p>
018     * 
019     * @param  args  Command line parameters.
020     */
021    public static void main(String[] args) {
022        CommandLineInterface cli = new CommandLineInterface();
023
024        if (args.length > 0) {
025            if (args[0].equals("gui")) {
026                try {
027                    Application.launch(GraphicalUserInterface.class);
028                } catch (UnsupportedOperationException e) {
029                    e.printStackTrace();
030                }
031            } else if (args[0].equals("cli")) {
032                cli.run();
033            }
034        } else {
035            try {
036                Application.launch(GraphicalUserInterface.class);
037            } catch (UnsupportedOperationException e) {
038                e.printStackTrace();
039                cli.run();
040            }
041        }
042    }
043}
