001import java.util.Scanner;
002
003/**
004 * CommandLineInterface is a simplified access point for MasterShopper9000
005 *
006 * @author      Jyri Virtaranta jyri.virtaranta@cs.tamk.fi
007 * @version     2017.11.05
008 * @since       1.8
009 */
010public class CommandLineInterface {
011    private ShoppingListApp shoppingListApp;
012    private Scanner scanner;
013    private boolean isRunning = true;
014
015    /**
016     * Instantiates a new Command line interface.
017     */
018    public CommandLineInterface() {
019        shoppingListApp = new ShoppingListApp(true);
020        scanner = new Scanner(System.in);
021    }
022
023    /**
024     * Returns isRunning boolean.
025     *
026     * @return  boolean  true if command line interface is running
027     */
028    public boolean isRunning() {
029        return isRunning;
030    }
031
032    /**
033     * Runs the command line interface.
034     */
035    public void run() {
036        String command;
037
038        System.out.println("SHOPPING LIST");
039        System.out.println("Tampere University of Applied Sciences");
040
041        while (isRunning) {
042            System.out.println("Give shopping list "
043                            + "(example: 1 milk;2 tomato;3 carrot;)");
044            command = scanner.nextLine().toLowerCase();
045
046            if (command.equals("exit") ||
047                    command.equals("quit")) {
048                stop();
049            }
050
051//            if (command.equals("save")) {
052//                shoppingListApp.saveToFile("shoppingList.txt");
053//            }
054
055//            if (command.equals("load")) {
056//                shoppingListApp.loadFromFile("shoppingList.txt");
057//                showListContent();
058//            }
059
060            if (shoppingListApp.inputChecker(command)) {
061                shoppingListApp.inputHandler(command);
062                showListContent();
063            }
064        }
065    }
066
067    /**
068     * Stops the command line interface
069     */
070    public void stop() {
071        shoppingListApp.close();
072        System.exit(0);
073    }
074
075    /**
076     * Prints shoppingList contents to console
077     */
078    private void showListContent() {
079        System.out.println("Your Shopping List now:");
080
081        for (int i = 0; i < shoppingListApp.getShoppingList().size(); i++) {
082            ShoppingItem listItem = shoppingListApp.getShoppingList().get(i);
083            String output = "  ";
084
085            output += listItem.getAmount();
086            output += " ";
087            output += listItem.getName();
088
089            System.out.println(output);
090        }
091
092        System.out.println();
093    }
094}
